class ExerciseSelectionEngine {
  List<Exercise> selectExercises({
    required String muscle,
    required int targetSets,
    required ClientProfile profile,
    required List<Exercise> availableExercises,
  }) {
    // 1. Filtrar por equipamiento disponible
    var candidates = availableExercises.where(
      (e) => profile.availableEquipment.containsAll(e.equipment)
    ).toList();
    
    // 2. Excluir ejercicios de riesgo por lesiones
    candidates = candidates.where(
      (e) => !_isContraindicated(e, profile.injuries)
    ).toList();
    
    // 3. Evaluar y rankear por matriz de criterios
    final ranked = candidates.map((e) {
      return ExerciseScore(
        exercise: e,
        score: _evaluateExercise(e, muscle, profile),
      );
    }).toList()..sort((a, b) => b.score.compareTo(a.score));
    
    // 4. Selecci√≥n estratificada
    final selected = <Exercise>[];
    int setsAssigned = 0;
    
    // Paso 4a: Compuestos fundamentales (60% volumen)
    final compounds = ranked.where((e) => 
      e.exercise.category == 'compound' && e.score >= 4.0
    ).take(2).toList();
    
    for (var compoundScore in compounds) {
      int sets = (targetSets * 0.30).round();
      selected.add(compoundScore.exercise.copyWith(sets: sets));
      setsAssigned += sets;
    }
    
    // Paso 4b: Aislados regionales (40% volumen)
    final isolations = ranked.where((e) => 
      e.exercise.category == 'isolation' && e.score >= 3.0
    ).take(3).toList();
    
    final remainingSets = targetSets - setsAssigned;
    final setsPerIsolation = (remainingSets / isolations.length).round();
    
    for (var isoScore in isolations) {
      selected.add(isoScore.exercise.copyWith(sets: setsPerIsolation));
    }
    
    return selected;
  }
  
  double _evaluateExercise(Exercise e, String muscle, ClientProfile profile) {
    // Aplicar matriz de ponderaci√≥n
    final scores = {
      'resistance_curve': _scoreResistanceCurve(e),
      'angle_length': _scoreAngleLength(e, muscle),
      'rom': _scoreROM(e),
      'progressive_overload': _scoreOverload(e),
      'complexity': _scoreTechnicalComplexity(e, profile.trainingLevel),
      'joint_stress': _scoreJointStress(e, profile.injuries),
    };
    
    final weights = {
      'resistance_curve': 0.15,
      'angle_length': 0.25,
      'rom': 0.20,
      'progressive_overload': 0.25,
      'complexity': 0.10,
      'joint_stress': 0.05,
    };
    
    double totalScore = 0.0;
    scores.forEach((criterion, score) {
      totalScore += score * weights[criterion]!;
    });
    
    return totalScore;
  }
}
